async function renderDashboard(code) { // 1. Force cleanup of existing instances ['priceChart', 'ccassChart'].forEach(id => { const chart = Chart.getChart(id); if (chart) chart.destroy(); }); document.getElementById('display-title').innerText = `${stocksIndex[code].name} (${code})`; const [priceData, ccassData] = await Promise.all([ fetchCSV(`price_data/${code}.csv`), fetchCSV(`ccass_pivoted/${code}.csv`) ]); if (!priceData || !ccassData) { console.error("Missing data for alignment"); return; } // 2. Map and Sort Data const ohlc = priceData.map(d => ({ x: luxon.DateTime.fromISO(d.date || d.Date).valueOf(), o: d.open || d.Open, h: d.high || d.High, l: d.low || d.Low, c: d.close || d.Close })).filter(d => d.x && !isNaN(d.c)).sort((a, b) => a.x - b.x); const brokers = Object.keys(ccassData[0]).filter(k => k !== 'date'); const colors = ['#3b82f6', '#ef4444', '#10b981', '#f59e0b', '#8b5cf6', '#ec4899', '#64748b', '#06b6d4', '#f97316', '#1e40af']; const ccassMapped = ccassData.map(d => ({ date: luxon.DateTime.fromISO(d.date).valueOf(), values: d })).filter(d => d.date).sort((a, b) => a.date - b.date); // 3. FIND COMMON DATE RANGE (CRITICAL FOR ALIGNMENT) const minDate = Math.max(ohlc[0].x, ccassMapped[0].date); const maxDate = Math.min(ohlc[ohlc.length - 1].x, ccassMapped[ccassMapped.length - 1].date); // 4. Shared X-Axis Config const sharedXAxis = { type: 'time', time: { unit: 'day' }, min: minDate, max: maxDate, grid: { display: false } }; // 5. Render Price Chart new Chart('priceChart', { type: 'candlestick', data: { datasets: [{ label: 'Price', data: ohlc, color: { up: '#10b981', down: '#ef4444' }, borderColor: { up: '#10b981', down: '#ef4444' } }] }, options: { responsive: true, maintainAspectRatio: false, scales: { x: sharedXAxis, y: { position: 'right' } }, plugins: { legend: { display: false } } } }); // 6. Render CCASS Chart const datasets = brokers.map((b, i) => ({ label: `${b} (${(ccassData[ccassData.length-1][b] || 0).toFixed(2)}%)`, data: ccassMapped.map(d => ({ x: d.date, y: d.values[b] })), backgroundColor: colors[i % colors.length] + 'aa', borderColor: colors[i % colors.length], fill: true, pointRadius: 0, borderWidth: 1 })); new Chart('ccassChart', { type: 'line', data: { datasets }, options: { responsive: true, maintainAspectRatio: false, scales: { x: sharedXAxis, // Use the exact same axis config y: { stacked: true, beginAtZero: true, ticks: { callback: v => v + '%' } } }, plugins: { legend: { position: 'bottom', labels: { boxWidth: 8, font: { size: 10 } } }, tooltip: { mode: 'index', intersect: false } } } }); }
